#smallpox discovery

namespace = small_pox_discovery

country_event = {
        id = small_pox.1
		title = "EVTNAME2000.1"                                     #A new Disease
		desc = "EVTDESC2000.1"
		picture = PLAGUE_eventPicture
		fire_only_once = yes
		
		trigger = {
		        has_disaster = small_pox		   
				NOT = {
				        owner = { has_culture_group = european }
				}		
				OR = {
				        has_technology_group = andean
                        has_technology_group = north_american
                        has_technology_group = mesoamerican
				}
				any_country = {
				        technology_group = western
						has_discovered = ROOT
						}
	    }		
		
		mean_time_to_happen = {
		        months = 12
		}
		
		option = {
		        name = "EVTOPTA2000.1"
				add_treasury = -102.40				
		}

		option = {
		        name = "EVTOPTB2000.1"
                modifier = manpower_smallpox_effect
                add_manpower = -2000
                add_prestige = -5								
		}
		
		option = {
		        name = "EVTOPTC2000.1"
                add_stability = -2
                add_prestige = -20	
                modifier = smallpox_depression	
                }				
}

#Success In Trading - Natives	

name_space = natives-success_in_trading_furs

country_modifier = {
        id = nativetrading.1
		title = "EVTNAME100.1"                                      #Fur-trade_control.
		desc = "EVTDESC100.1"
		event_picture = NATIVETRADINGSUCCESS_eventPicture
		
		trigger = {
		        NOT = { has_disaster = small_pox }
				AND = {
				        stability = 1
						technology_group = andean
						technology_group = mesoamerican
						technology_group = north_american
						production_leader = {
						        trade_goods = fur
								value = yes
						}
                }
        }

        mean_time_to_happen = { days = 30 }

        option = {
		        name = "EVTOPTA100.1"                              #This will strengthen our power.
                if = {
				        limit = {
                                prestige = 50
								stability = 2
                        }
                        add_adm_points = 10
                 }
                add_country_modifier = {
				                name = success_in_trading
								duration = 60
				}
        }
}
